**backend/app/services/email_service.py** — Gmail service:

1. `async def list_gmail_threads(user_id: str, max_results: int = 20) -> list[dict]`:
   - Get valid Google token via get_valid_token(user_id)
   - GET https://gmail.googleapis.com/gmail/v1/users/me/threads with:
     - Authorization: Bearer {token}
     - params: maxResults={max_results}, labelIds=INBOX
   - For each thread in response: GET .../threads/{thread_id}?format=metadata&metadataHeaders=Subject&metadataHeaders=From&metadataHeaders=Date
   - Extract: thread_id (id), subject (from Subject header), sender (from From header), date (from Date header of first message), snippet, message_count (len(messages))
   - Return list sorted by date descending
   - On 401: raise HTTPException with "Google token expired, please reconnect"

2. `async def get_gmail_thread(user_id: str, thread_id: str) -> dict`:
   - Get valid Google token
   - GET https://gmail.googleapis.com/gmail/v1/users/me/threads/{thread_id}?format=full
   - For each message: extract from, date, subject, and body
   - Body extraction: prefer text/plain part; if only text/html, strip HTML tags using a simple regex or html.parser
   - Decode base64url-encoded body parts
   - Return GmailThreadDetail dict

3. `def normalise_email_thread(thread_detail: dict) -> str`:
   - Concatenate messages with headers between them:
     ```
     From: sender@email.com
     Date: 2026-03-04 10:30 AM
     Subject: Re: Project Update

     Message body text here...

     ---

     From: other@email.com
     Date: 2026-03-04 09:15 AM
     Subject: Project Update

     Original message body...
     ```
   - This normalised text becomes the raw_transcript for the meeting record
   - Strip excessive whitespace, normalise line endings

4. `async def import_email_as_meeting(user_id: str, thread_id: str, provider) -> str`:
   - Fetch thread detail via get_gmail_thread
   - Normalise to transcript text via normalise_email_thread
   - Create a title from subject: "Email: {subject}"
   - Extract date from first message
   - Create meeting record in DB:
     ```python
     supabase.table("meetings").insert({
         "user_id": user_id,
         "title": f"Email: {subject}",
         "meeting_date": extracted_date,
         "meeting_type": "email_thread",
         "raw_transcript": normalised_text,
         "status": "pending",
     }).execute()
     ```
   - DO NOT upload to Supabase Storage (no file to store — text is inline)
   - Return the new meeting_id
   - The AI processing will be triggered in the API endpoint using BackgroundTasks

All endpoints require auth.

```python
@router.get("/threads")
async def list_threads(
    user: AuthenticatedUser = Depends(get_current_user),
    max_results: int = 20,
):
    threads = await list_gmail_threads(user.id, max_results)
    return {"threads": threads}

@router.get("/threads/{thread_id}")
async def get_thread(
    thread_id: str,
    user: AuthenticatedUser = Depends(get_current_user),
):
    thread = await get_gmail_thread(user.id, thread_id)
    return thread

@router.post("/threads/{thread_id}/import", status_code=202)
async def import_thread(
    thread_id: str,
    background_tasks: BackgroundTasks,
    user: AuthenticatedUser = Depends(get_current_user),
    provider: LLMProvider = Depends(get_llm_provider),
):
    meeting_id = await import_email_as_meeting(user.id, thread_id, provider)

    # Fetch the normalised transcript for processing
    supabase = get_supabase_client()
    meeting = supabase.table("meetings").select("raw_transcript").eq("id", meeting_id).execute()
    transcript_text = meeting.data[0]["raw_transcript"]

    # Trigger full AI processing pipeline
    background_tasks.add_task(
        process_meeting,
        meeting_id=meeting_id,
        transcript_text=transcript_text,
        provider=provider,
        user_id=user.id,
    )

    return EmailImportResponse(meeting_id=meeting_id, status="pending")
```

// Gmail
export async function getGmailThreads(token: string): Promise<GmailThread[]> {
  const res = await fetch(`${API_URL}/api/v1/email/threads`, {
    headers: { Authorization: `Bearer ${token}` },
  })
  if (!res.ok) {
    const err = await res.json().catch(() => ({}))
    throw new Error(err.detail || "Failed to fetch email threads")
  }
  const data = await res.json()
  return data.threads
}

export async function getGmailThread(threadId: string, token: string): Promise<GmailThreadDetail> {
  const res = await fetch(`${API_URL}/api/v1/email/threads/${threadId}`, {
    headers: { Authorization: `Bearer ${token}` },
  })
  if (!res.ok) {
    const err = await res.json().catch(() => ({}))
    throw new Error(err.detail || "Failed to fetch email thread")
  }
  return res.json()
}

export async function importEmailThread(threadId: string, token: string): Promise<MeetingUploadResponse> {
  const res = await fetch(`${API_URL}/api/v1/email/threads/${threadId}/import`, {
    method: "POST",
    headers: { Authorization: `Bearer ${token}` },
  })
  if (!res.ok) {
    const err = await res.json().catch(() => ({}))
    throw new Error(err.detail || "Failed to import email thread")
